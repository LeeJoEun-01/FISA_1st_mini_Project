JavaScript *
- 브라우저 내에서 실행되는 인터프리터 언어
; <- 필수

1. 타입
	name = '정이';
	name = "정이";
	타입 확인 - string
	
	var name = '정이';
	let name = '정이';
	const name = '정이';
		
	name = 3;
	타입 확인 - number 타입으로 확인
